// sketch.js let x = 100, y = 100; function setup() { createCanvas(400, 400); // 400x400 캔버스 만들기 } function draw() { background(220); // 회색 배경 rect(x, y, 50, 50); // x, y에 50x50 정사각형 그리기 } function keyPressed() { if (keyCode === LEFT_ARROW) x -= 10; else if (keyCode === RIGHT_ARROW) x += 10; else if (keyCode === UP_ARROW) y -= 10; else if (keyCode === DOWN_ARROW) y += 10; }
